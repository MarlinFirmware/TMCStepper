28uint8_t SW_SPIClass::transfer(uint8_t ulVal) {

31
32 for (uint8_t i=7 ; i>=1 ; i--) {
33 // Write bit
34 !!(ulVal & (1 << i)) ? writeMOSI_H : writeMOSI_L;
